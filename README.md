# Node-Intern


# To setUp database create a .env file and add the below details
PORT=8080 
PHP='Your database name'
DB_USER= username
DB_PASS=password
DB_HOST=hostname
DB_PORT=db port number

# to run the api use nodemon server.js in the project folder
